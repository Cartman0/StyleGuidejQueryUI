addClass

.addClass( className [, duration ] [, easing ] [, complete ] )

easing でアニメーション(effect)を付与しながら、classを追加できる。



<div class="box div-addClass"></div>





.effeccts-addClass-box {
  width:100px;
  height:100px;
  border: 1px solid #333;
}


.effeccts-addClass__big-danger {
  background-color: #c33;
  width: 200px;
  height: 200px;
}





$('.effeccts-addClass-box').click( function() {
  $(this).addClass('effeccts-addClass__big-danger', 1000, "easeOutBounce" );
});

removeClass

.removeClass( className [, duration ] [, easing ] [, complete ] )

easing でアニメーション(effect)を付与しながら、classを削除できる。



<div class="effeccts-removeClass-box effeccts-removeClass__big-info"></div>





.effeccts-removeClass-box {
  width:100px;
  height:100px;
  border: 1px solid #333;
}


.effeccts-removeClass__big-info {
  background-color: #37c;
  width: 200px;
  height: 200px;
}





$('.effeccts-removeClass-box').click(function() {
  console.log(1);
  $( this ).removeClass('effeccts-removeClass__big-info', 1000, 'easeInBack');
});
