Color Animation

backgroundColor

click


<div class="animate-box animate-backgroundColor">click</div>





.animate-box {
  width: 100px;
  height: 100px;
  border: solid 1px #333;
}





$('.animate-backgroundColor').click(function() {
  $(this).animate({
    backgroundColor: 'rgb( 20, 20, 20 )'
  });
});
