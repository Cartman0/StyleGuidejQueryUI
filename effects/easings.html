easings

linear

code 


<div class="easings-box">
  <div class="easings-ball easings-linear"></div>
</div>





.easings-box{
  position: relative;
  width: 100%;
  height: 50px;
  border: 1px solid #333;
  box-sizing: border-box;
  padding: 0 5px;
  margin: 0;
}


.easings-ball{
  position: absolute;
  width: 50px;
  height: 50px;
  border: 1px solid #333;
  border-radius: 50%;
  background-color: #d33;
  margin: 0;
  padding: 0;
  box-sizing: border-box;
}





$('.easings-linear').click(function() {
  $.when(
    $(this).animate({
      left: $(this).parent().width() - $(this).width() + 10
    }, 1000, 'linear')
  )
  .done(function(){
    $(this).animate({
      left: 5
    }, 1000, 'linear');
  });
});
