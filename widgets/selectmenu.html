var circle = $('.selectmenu-circle');


$('#selectmenu-radius').selectmenu({
  width: 200,
  change: function( event, data ) {
    circle.css({
      width: data.item.value,
      height: data.item.value
    });
  }
 });


$('#selectmenu-color').selectmenu({
  width: 200,
  change: function( event, data ) {
    circle.css( "background", data.item.value );
  }
 });
