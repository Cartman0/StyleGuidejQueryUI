Options

nooption

HTML5 default: <input type="date">
jQueryUI datepicker


<button class="button-nooption">button</button>


<a class="button-nooption">a</a>





$('.button-nooption').button();

altField, altFormat: 別のFieldに、別のFormatに出力



<input type="text" class="datepicker-alt">


<input type="text" class="datepicker-alt-alternate">





$('.datepicker-alt').datepicker({
  altField: ".datepicker-alt-alternate",
  altFormat: "DD, d MM, yy" /* ex. Tuesday, 4 August, 2015 */
});

appendText: textが挿入される



<input type="text" class="datepicker-appendText">





$('.datepicker-appendText').datepicker({
  appendText: '[mm/dd/yyyy]'
});

autoSize



<input type="text" class="datepicker-autoSize">





$('.datepicker-autoSize').datepicker({
  autoSize: true
});

beforeShow: カレンダーがレンダリングされる直前に呼び出される + getDate

getDate: 


<div class="div-datepicker-beforeShow-getDate">getDate: </div>
<input type="text" class="datepicker-beforeShow">





$(".datepicker-beforeShow").datepicker({
  // カレンダーがレンダリングされる直前に呼び出される
  beforeShow: function(input, inst) {
    // テキストフィールドに値が入っていればその日付を、入っていなければ今日の日付を取得
    var date = $(input).datepicker("getDate");
    $('.div-datepicker-beforeShow-getDate').text('getDate: ' + date);
  }
});

changeMonth: month を dropbox で変えられる



<input type="text" class="datepicker-changeMonth">





$('.datepicker-changeMonth').datepicker({
  changeMonth: true
});

changeYear: yearRange を指定しなければならない



<input type="text" class="datepicker-changeYear">





$('.datepicker-changeYear').datepicker({
  changeYear: true,
  yearRange: '2010:2020'
});

dateFormat: 日付の表示のフォーマット変更, default: "mm/dd/yy"



<input type="text" class="datepicker-dataFormat">





$('.datepicker-dataFormat').datepicker({
  dateFormat: "yy-mm-dd"
});

dayNamesMin: datepicker上の曜日名を変更



<input type="text" class="datepicker-dayNamesMin">

maxDate, minDate: 有効範囲を決める

2週間前から1ヶ月後まで
