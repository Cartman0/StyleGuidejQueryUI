Methods

value()



<div class="slider-methods-value"></div>


<button class="btn-slider-methods-value">Get</button>
<input type="number" class="input-slider-methods-value">





$('.slider-methods-value').slider();


$('.btn-slider-methods-value')
.button()
.click(function() {
  var v = $('.slider-methods-value').slider( "value" );
  $('.input-slider-methods-value').val(v);
});

Events

change: slideして、マウスのドラッグが終わり、値が決定したらイベント発生



<div class="slider-change"></div>


<input type="number" class="input-slider-change">





$('.slider-change').slider({
  change: function(event, ui){
    $('.input-slider-change').val(ui.value);
  }
});

slide( event, ui ): slide している間、イベント発生



<div class="slider-slide"></div>
<input type="number" class="input-slider-slide">





$('.slider-slide').slider({
  slide: function(event, ui){
    $('.input-slider-slide').val(ui.value);
  }
});

<div class="slider-progressbar-progressbarandslider" style="margin: 1rem;"></div>





$('.slider-slider-progressbarandslider').slider({
  slide: function(event, ui){
    $('.slider-progressbar-progressbarandslider').progressbar( 'option', 'value', ui.value);
  }
});
