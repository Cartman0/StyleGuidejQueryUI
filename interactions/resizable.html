Events

resize( event, ui )

width: height:
stop


<div class="div-resize">width: height:</div>


<div class="resize" style="width: 100px; height: 100px; border: 1px solid #333;">stop</div>





$('.resize').resizable({
  resize: function( event, ui ) {
    $('.div-resize').text('width:' + ui.size.width + ' height:' + ui.size.height);
  }
});
